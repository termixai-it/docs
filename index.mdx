## Example Implementation

Here's what a basic implementation looks like:

```tsx
import { ChatInterface } from '@termix-it/react-tool';
import '@termix-it/react-tool/dist/styles.css';

function App() {
  return (
    <div className="h-screen">
      <ChatInterface
        projectId="your-project-id"
        aiConfigId="your-ai-config-id"
        apiBaseUrl="/api/proxy"
        authorization="Bearer your-jwt-token"
        enableFunctionCalls={true}
        enableKnowledgeBase={true}
        onFunctionExecuted={(call, result) => {
          console.log('Function executed:', call.name, result);
        }}
      />
    </div>
  );
}
```
